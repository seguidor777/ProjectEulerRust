Factorized number providing multiple or divide operation without causing overflow.

Example

use prime::{Factorized, PrimeSet};
use std::iter;

// Calculates 40C20
let ps = PrimeSet::new();
let mut fac = Factorized::<u64>::new(&ps);
for n in 21..41 {
    fac.mul_assign(n);
}
for n in 1..21 {
    fac.div_assign(n);
}
assert_eq!(137846528820, fac.into_integer());
