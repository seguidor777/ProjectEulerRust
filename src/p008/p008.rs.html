fn compute(prod_len: usize) -> u64 {
    INPUT
        .chars()
        .filter_map(|c| c.to_digit(10))
        .map(|n| n as u64)
        .collect::<Vec<_>>()
        .windows(prod_len)
        .map(|win| win.iter().fold(1u64, |p, &n| p * n))
        .max()
